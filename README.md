% Trust hierarchy
agent(architect, high,   governance).
agent(auditor,   high,   compliance).
agent(engineer,  medium, execution).
agent(diplomat,  medium, negotiation).
agent(custodian, low,    maintenance).
agent(sentinel,  high,   security).

% Action requirements  [name, min_trust, domain, resonance_floor]
action(governance_vote,  high,   governance,   93).
action(audit_seal,       high,   compliance,   87).
action(deploy_contract,  medium, execution,    75).

action(archive_record,   low,    maintenance,  65).
action(raise_alert,      high,   security,     91).
action(read_archive,     low,    any,          60).
action(submit_proposal,  medium, any,          70).

% Inference rules
trust_gte(high, high).   trust_gte(high, medium).   trust_gte(high, low).
trust_gte(medium, medium). trust_gte(medium, low).   trust_gte(low, low).
domain_ok(_, any).       domain_ok(D, D).
resonance_ok(H) :- H >= 60.

% The gate
can_execute(A, X) :-
  agent(A, T, D),
  action(X, RT, RD, H),
  trust_gte(T, RT),
  domain_ok(D, RD),
  resonance_ok(H).

% ?- can_execute(custodian, governance_vote).   → FALSE (trust low < high)
% ?- can_execute(engineer,  audit_seal).        → FALSE (domain execution ≠ compliance)
